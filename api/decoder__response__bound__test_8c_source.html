00038 void snap_test_decode_response_bound_imp(size_t size)
00039 {
00040     snap_t snap;
00041     snap_init( &snap );
00042 
00043     size_t addresses[] = { 0, 1, 5, 128, 255, 256, 32434, 65535, 65536, 16777215 };
00044     const size_t addresses_size = sizeof( addresses ) / sizeof( addresses[ 0 ] );
00045 
00046     for(size_t i = 0; i < addresses_size; ++i) {
00047         for(size_t k = 0; k < addresses_size; ++k) {
00048             snap_reset( &snap );
00049 
00050             SNAP_ASSERT( snap_is_ready( &snap ) );
00051 
00052             snap_set_local_address( &snap, addresses[ i ] );
00053             snap_set_peer_address( &snap, addresses[ k ] );
00054 
00055             if( size < 3 ) {
00056                 SNAP_ASSERT( snap_decode_response_bound( &snap, size ) == 0 );
00057             }
00058             else {
00059                 // NOTICE
00060                 // currently snap_util_ack_response_bound and snap_util_nak_response_bound both
00061                 // returns 4 (3 bytes header + 1 byte checksum).
00062 
00063                 const size_t expected = 4 + snap_util_get_address_size( addresses[ i ] )
00064                                           + snap_util_get_address_size( addresses[ k ] );
00065 
00066                 SNAP_ASSERT( snap_decode_response_bound( &snap, size ) == expected );
00067 
00068                 // with bound (e.g. snap waits for ack );
00069                 size_t bound[] = {
00070                     3, 4, 5, 6, 7, 8, 9, 10, 15, 16, 17, 18, 19, 20,  31, 32,
00071                     33, 63, 64, 65, 127, 128, 129, 254, 255, 256, 300, 511, 512
00072                 };
00073 
00074                 for(size_t i = 0; i < sizeof( bound ) / sizeof( bound[ 0 ] ); ++i) {
00075                     snap.packet.size = 43;
00076                     SNAP_ASSERT( !snap_is_ready( &snap ) );
00077 
00078                     snap.packet.bound = bound[ i ];
00079 
00080                     SNAP_ASSERT( snap_decode_response_bound( &snap, size ) == bound[ i ] );
00081                 }
00082             }
00083         }
00084     }
00085 }
