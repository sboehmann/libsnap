00103 uint8_t snap_crc8(const uint8_t *data, size_t size)
00104 {
00105     uint8_t value = CRC8_INIT;
00106 
00107     for(size_t i = 0; i < size; ++i) {
00108         uint8_t byte = data[ i ];
00109 
00110         for(size_t k = 8; k > 0; --k) {
00111             bool feedback_bit = ( ( value ^ byte ) & 0x01 ) == 0x01;
00112 
00113             if( feedback_bit ) {
00114                 value ^= CRC8_POLYNOMIAL;
00115             }
00116             value = (value >> 1) & 0x7F;
00117 
00118             if( feedback_bit ) {
00119                 value |= 0x80;
00120             }
00121 
00122             byte = byte >> 1u;
00123         }
00124     }
00125
