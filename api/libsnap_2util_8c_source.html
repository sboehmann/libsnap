00425 bool snap_util_priv_create_ack_nak_packet(PacketType type, uint32_t local_address, uint32_t peer_address, uint8_t *data, size_t *pos, size_t size)
00426 {
00427     SNAP_CHECK_PTR( data );
00428 
00429     SNAP_ASSERT( snap_util_ack_response_bound(local_address, peer_address ) == snap_util_nak_response_bound(local_address, peer_address ) );
00430 
00431     if( ( size - *pos ) < snap_util_ack_response_bound(local_address, peer_address ) ) {
00432         return false;
00433     }
00434 
00435     const size_t local_add_size = snap_util_get_address_size( local_address );
00436     const size_t peer_add_size = snap_util_get_address_size( peer_address );
00437 
00438     const size_t initialPos = *pos;
00439 
00440     data[ (*pos)++ ] = snap_util_get_sync_byte();
00441     data[ (*pos)++ ] = ( peer_add_size << 6 ) + ( local_add_size << 4 ) + (uint8_t) type;
00442     data[ (*pos)++ ] = 32; /* Only bit 5 (e.g. edm is 8 bit checksum) */
00443 
00444     switch( peer_add_size ) {
00445     case 3:
00446         data[ (*pos)++ ] = (uint8_t) ( peer_address >> 16 );
00447     case 2:
00448         data[ (*pos)++ ] = (uint8_t) ( peer_address >> 8 );
00449     case 1:
00450         data[ (*pos)++ ] = (uint8_t) ( peer_address );
00451     default:
00452         break;
00453     }
00454 
00455     switch( local_add_size ) {
00456     case 3:
00457         data[ (*pos)++ ] = (uint8_t) ( local_address >> 16 );
00458     case 2:
00459         data[ (*pos)++ ] = (uint8_t) ( local_address >> 8 );
00460     case 1:
00461         data[ (*pos)++ ] = (uint8_t) ( local_address );
00462     default:
00463         break;
00464     }
00465 
00466     data[ *pos ] = snap_checksum( &data[ initialPos ], (*pos) - initialPos );
00467     (*pos)++;
00468 
00469     return true;
00470 }
