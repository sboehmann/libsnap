00459     if( in_data_size < 3 ) {

00462

00464         // the bound of the last packet can't be less then 3 (sync + hdb2 + hdb1 )

00635     for(uint8_t i = dab; i > 0; --i) {
00636         dest_address += ( in_data[ real_pos++ ] << ( ( i - 1u ) * 8u ) );
00637     }
00638 
00639     // read source address bytes, if any
00640     uint32_t source_address = 0;
00641     for(uint8_t i = sab; i > 0; --i) {
00642         source_address += ( in_data[ real_pos++ ] << ( ( i - 1u ) * 8u ) );
00643     }
00644 
00645 #ifdef SNAP_REJECT_PACKETS_WITH_WRONG_ADDRESS
00646     // TODO Test (i.e. write unit tests) !!!
00647     // if we have a real destination address (0 means broadcarst or no address
00648     // at all) then check if we are the receiver.
00649     if( dest_address != 0 && snap_get_local_address( ptr ) != 0 ) {
00650         if( snap_get_local_address( ptr ) != dest_address ) {
00651             // We are not the receiver of ptr data packet, so skip ptr packet.
00652             *in_data_pos += end_of_data;
00653 
00654             return true;
00655         }
00656     }
00657 #endif
00658 
00659     // calculate current packet checksum
00660     uint32_t edm_code_expected = 0;
00661     switch( edm ) {
00662         case EDM_THREETIMES:
00663         case EDM_FEC:
00664         case EDM_USER:
00665             break;
00666 
00667         case EDM_CHECKSUM:
00668             edm_code_expected = snap_checksum( in_data + begin_of_packet + 1,
00669                                             end_of_data - 1
00670                                              );
00671             break;
00672 
00673         case EDM_CRC8:
00674             edm_code_expected = snap_crc8( in_data + begin_of_packet + 1,
00675                                             end_of_data - 1
00676                                          );
00677             break;
00678 
00679         case EDM_CRC16:
00680             edm_code_expected = snap_crc16( in_data + begin_of_packet + 1,
00681                                             end_of_data - 1
00682                                            );
00683             break;
00684 
00685         case EDM_CRC32:
00686             edm_code_expected = snap_crc32( in_data + begin_of_packet + 1,
00687                                             end_of_data - 1
00688                                           );
00689             break;
00690 
00691         default: // (No EDM)
00692             SNAP_ASSERT( edm == EDM_NONE );
00693             break;
00694     }
00695 
00696     // check error detection code
00697     SNAP_ASSERT( end_of_packet - end_of_data == edm_size );
00698 
00699     uint32_t edm_code = 0;
00700     size_t edm_pos = end_of_data;
00701     for(size_t i = edm_size; i > 0; --i) {
00702         edm_code += in_data[ edm_pos++ ] << ( ( i - 1u ) * 8u );
00703     }
00704 
00705     if( edm_code != edm_code_expected ) {
00706         ptr->last_decode_error_pos = end_of_data;
00707         ptr->last_error = SNAP_ERROR_INVALID_EDM;
00708 
00709         *in_data_pos += end_of_packet;
00710 
00711         if( ack == 1 ) {
00712             snap_util_create_nak_response( dest_address, source_address,
00713                                          response, response_pos, response_size
00714                                        );
00715         }
00716 
00717         return false;
00718     }
00719 
00720     // calculate number of fill bytes (0-bytes) we can/have to skip from the end.
00721     size_t skip_byte_count = 0;
00722     for(size_t i = end_of_data - 1; i > begin_of_data; --i) {
00723         if( in_data[ i ] != snap_util_get_fill_byte() ) {
00724             break;
00725         }
00726 
00727         skip_byte_count++;
00728     }
00729 
00730     // read significant data bytes
00731     for(size_t i = begin_of_data; i < end_of_data - skip_byte_count; ++i) {
00732         uint8_t value = in_data[ i ];
00733         result[ *result_pos ] = value;
00734         (*result_pos)++;
00735         real_pos++;
00736     }
00737 
00738     if( ack == 1 ) {
00739         snap_util_create_ack_response( dest_address, source_address,
00740                                      response, response_pos, response_size
00741                                    );
00742     }
00743 
00744     if( edm == EDM_THREETIMES ) {
00745         *in_data_pos += 3 * end_of_packet;
00746     }
00747     else {
00748         *in_data_pos += end_of_packet;
00749     }
00750 
00751     return true;
00752 }
